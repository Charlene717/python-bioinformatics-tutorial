import torch
import torch.nn as nn

# Tensor = 帶 GPU 加速的 NumPy ndarray
x = torch.randn(3, 4)         # shape (3,4) on CPU
x.shape; x.dtype          # torch.float32

# 移到 GPU
device = "cuda" if torch.cuda.is_available() else "cpu"
x = x.to(device)

# NumPy ↔ Tensor
import numpy as np
arr = np.random.rand(3,4)
t = torch.from_numpy(arr).float()
back = t.cpu().numpy()    # GPU tensor → CPU → numpy

# 自動微分 (這是深度學習的核心！)
x = torch.randn(3, requires_grad=True)
y = (x ** 2).sum()
y.backward()              # 計算 dy/dx
print(x.grad)             # 2x
# 一個極簡的「DNA 序列 → 表達量」CNN
class SeqCNN(nn.Module):
    def __init__(self, seq_len=200):
        super().__init__()
        self.conv1 = nn.Conv1d(4, 32, kernel_size=8, padding=4)
        self.conv2 = nn.Conv1d(32, 64, kernel_size=8, padding=4)
        self.pool  = nn.AdaptiveAvgPool1d(1)
        self.fc    = nn.Linear(64, 1)

    def forward(self, x):
        # x: (batch, 4, seq_len) — one-hot DNA
        x = torch.relu(self.conv1(x))
        x = torch.relu(self.conv2(x))
        x = self.pool(x).squeeze(-1)   # (batch, 64)
        return self.fc(x).squeeze(-1)   # (batch,)

model = SeqCNN().to(device)
print(model)
print(f"# params: {sum(p.numel() for p in model.parameters()):,}")
import torch.optim as optim
from torch.utils.data import DataLoader, TensorDataset

# 假資料：1000 條 200bp 的 DNA → 表達量
N, L = 1000, 200
X = torch.randn(N, 4, L)                  # one-hot 假資料
y = torch.randn(N)
loader = DataLoader(TensorDataset(X, y), batch_size=32, shuffle=True)

opt = optim.Adam(model.parameters(), lr=1e-3)
loss_fn = nn.MSELoss()

for epoch in range(10):
    total = 0
    for xb, yb in loader:
        xb, yb = xb.to(device), yb.to(device)
        pred = model(xb)
        loss = loss_fn(pred, yb)
        opt.zero_grad(); loss.backward(); opt.step()
        total += loss.item() * xb.size(0)
    print(f"epoch {epoch}: {total/N:.4f}")

# 推論
model.eval()
with torch.no_grad():
    out = model(X[:5].to(device))

# pip install transformers torch
from transformers import AutoTokenizer, AutoModel
import torch

# ESM-2 (650M parameters) 來自 HuggingFace
tokenizer = AutoTokenizer.from_pretrained("facebook/esm2_t33_650M_UR50D")
model = AutoModel.from_pretrained("facebook/esm2_t33_650M_UR50D")

seqs = ["MKTAYIAKQRQISFVKSHFSRQLEERLGLIE",
        "MAGGSAALPDFAALADKKAEKMR"]

inputs = tokenizer(seqs, return_tensors="pt", padding=True)
with torch.no_grad():
    out = model(**inputs)

# 每個位置一個 1280-d embedding
print(out.last_hidden_state.shape)   # (2, max_len, 1280)

# 平均池化 → 一條序列一個向量 (用於分類 / 比對)
mask = inputs.attention_mask.unsqueeze(-1)
seq_emb = (out.last_hidden_state * mask).sum(1) / mask.sum(1)
# (2, 1280) — 拿來算相似度、做下游分類

# 應用：突變影響預測 / Zero-shot variant effect prediction
# ESM-2 可以對每個位置每個胺基酸算 log-likelihood，
# 高 = 該突變不影響蛋白；低 = 可能影響功能

# pip install transformers fair-esm
from transformers import EsmForProteinFolding, AutoTokenizer

tokenizer = AutoTokenizer.from_pretrained("facebook/esmfold_v1")

seq = "MKTAYIAKQRQISFVKSHFSRQLEERLGLIEVQKQHTHK"
inputs = tokenizer(seq, return_tensors="pt").to("cuda")

with torch.no_grad():

四、AlphaFold 用法

本機跑 AlphaFold 2 / 3 需要大量 GPU + 100+ GB 資料庫。新手請優先用以下三條路：

  • AlphaFold DB (alphafold.ebi.ac.uk)：兩億條已預測蛋白結構，用 UniProt ID 直接下載 PDB。
  • ColabFold：Google Colab 免費 GPU，貼序列就能跑。Python 介面：!pip install colabfold。
  • ESMFold (本章上面)：無需 MSA，速度比 AF2 快 60×，準確度略低但夠用。

Running AlphaFold 2 / 3 locally needs hefty GPUs and 100+ GB of databases. For starters, prefer:

  • AlphaFold DB (alphafold.ebi.ac.uk): 200M+ pre-predicted structures, download PDBs by UniProt ID.
  • ColabFold: free Google Colab GPU, paste a sequence to run. Python: !pip install colabfold.
  • ESMFold (above): no MSA, ~60× faster than AF2, slightly lower accuracy but often sufficient.
import requests
from pathlib import Path

def fetch_alphafold(uniprot_id, out_dir="results/structures"):
    url = f"https://alphafold.ebi.ac.uk/files/AF-{uniprot_id}-F1-model_v4.pdb"
    r = requests.get(url, timeout=30)
    r.raise_for_status()
    Path(out_dir).mkdir(parents=True, exist_ok=True)
    fp = Path(out_dir) / f"{uniprot_id}.pdb"
    fp.write_text(r.text)
    return fp

pdb = fetch_alphafold("P04637")         # 人類 TP53
print(f"Downloaded {pdb}")

# 用 Bio.PDB 讀進來
from Bio.PDB import PDBParser
parser = PDBParser(QUIET=True)
structure = parser.get_structure("TP53", pdb)
for chain in structure[0]:
    for residue in chain:
        if residue.id[1] == 175:           # TP53 R175H 熱點
            print(residue.get_resname(), residue.id)
