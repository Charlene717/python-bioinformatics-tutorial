四、ORF Finder

from Bio.Seq import Seq

def find_orfs(seq: Seq, min_aa: int = 100):
    """找出所有大於 min_aa 個 amino acid 的 ORF（六個讀框）。"""
    orfs = []
    for strand, s in [("+", seq), ("-", seq.reverse_complement())]:
        for frame in range(3):
            sub = s[frame:]
            # 翻譯整段，再用 * 切割
            prot = str(sub[: len(sub) - len(sub) % 3].translate())
            # 對每段 (用 stop 分隔)，再以 M 為起點
            for piece in prot.split("*"):
                if "M" in piece:
                    orf = piece[piece.index("M"):]
                    if len(orf) >= min_aa:
                        orfs.append((strand, frame, len(orf), orf))
    return orfs
